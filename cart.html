/* ========================= 🛒 GIỎ HÀNG LOCALSTORAGE ========================= */ document.addEventListener("DOMContentLoaded", () => { const cartListEl = document.getElementById("cartList"); const totalEl = document.getElementById("total"); const cartActionEl = document.getElementById("cartAction"); // --- Lấy giỏ hàng từ localStorage function getCart(){ return JSON.parse(localStorage.getItem("cart")) || []; } // --- Lưu giỏ hàng function saveCart(cart){ localStorage.setItem("cart", JSON.stringify(cart)); } // --- Tính tổng tiền function calculateTotal(cart){ return cart.reduce((sum, item) => sum + item.price * item.qty, 0); } // --- Render giỏ hàng function renderCart(){ const cart = getCart(); cartListEl.innerHTML = ""; cartActionEl.innerHTML = ""; if(cart.length === 0){ cartListEl.innerHTML = '

Giỏ hàng trống 😢

'; totalEl.textContent = ""; return; } cart.forEach(item => { const div = document.createElement("div"); div.className = "item"; div.innerHTML = `

${item.name}

${item.price.toLocaleString()}đ
${item.qty}
`; // tăng giảm số lượng div.querySelector(".minus").addEventListener("click", ()=>{ if(item.qty > 1){ item.qty--; saveCart(cart); renderCart(); } }); div.querySelector(".plus").addEventListener("click", ()=>{ item.qty++; saveCart(cart); renderCart(); }); // xóa sản phẩm div.querySelector(".remove").addEventListener("click", ()=>{ const index = cart.findIndex(i => i.id === item.id); if(index > -1){ cart.splice(index, 1); saveCart(cart); renderCart(); } }); cartListEl.appendChild(div); }); // hiển thị tổng tiền const total = calculateTotal(cart); totalEl.textContent = "Tổng: " + total.toLocaleString() + "đ"; // nút thanh toán cartActionEl.innerHTML = ``; } // --- Thanh toán (xóa giỏ hàng) window.checkout = function(){ if(confirm("Xác nhận thanh toán?")){ localStorage.removeItem("cart"); renderCart(); alert("Thanh toán thành công 🛒✅"); } } renderCart(); });
